الخلاصة

ُيعتبر قطــــاع صناعة منتجات حديد الصلــــب الطويلة عنصرًا حيويــــًا فــــي التنميــــة المحليــــة و التجــــارة الإقليميــــة على حد ســــواء، مع التركيز بشــــكل كبير على حديد التسليح و قضبان الأسلاك و المقاطع الفولاذية (التي تســــمى منتجات الحديد الطويلة.) و تلعب هذه الصناعة دورًا حيويًا في البنية التحتية و أعمال الإنشــــاءات، و التي تعتبر ضروريــــة للنمو الاقتصادي في الــــبلاد. و يخصص %89 مــــن منتجات الحديــــد الطويلة في الســــعودية لحديــــد التســــليح و قضبــــان الأسلاك، و التي تعتبــــر ضرورية لقطــــاع الإنشــــاءات، بينما تشــــكل المقاطع الفولاذيــــة النســــبة المتبقيــــة البالغــــة %11. و يشــــهد هــــذا القطاع نموًا كبيرًا بســــبب زيادة أنشطة الإنشاءات و مشاريع البنية التحتيــــة، مما يعكس على ســــبيل المثال معدل نمو ســــنوي مركب قوي يبلغ %8 لحديد التســــليح و %4 قضبان الاسلاك خلال الفتــــرة من 2018 إلى 2022. و على الصعيد العالمــــي، يشــــهد ســــوق منتجــــات الحديــــد الطويلة مســــارًا تصاعديًا بمعدل نمو ســــنوي مركب متوقع يبلغ %4.5.

و على الصعيد الإقليمي، شــــهد ســــوق الحديــــد و الصلب في الشرق الأوســــط تعافيًا وتوســــعًا مدفوعًا بالإنفاق الحكومي و جهود التنويــــع الاقتصادي على الرغم مــــن تداعيات جائحة كورونــــا، مــــن المتوقع أن ينمو ســــوق الحديــــد و الصلب في المملكة العربية الســــعودية بمعــــدل %3، مدعومًا بالتقدم التكنولوجي و زيادة قدرات الإنتاج المحلية. تعمل الشــــركات الرئيســــية فــــي المنطقة علــــى تعزيــــز قدرتها التنافســــية من خلال الابتكارات و التوسعات الاستراتيجية. و مع ذلك، يواجه القطــــاع تحديات مثل تقلب أســــعار المــــواد الخام و نقصها، خاصــــة خردة الحديد،مما يؤثر علــــى تكاليف الإنتاج و الأرباح. و توجد فــــرص كبيرة للنمــــو، و خاصة بفضل مشــــاريع البنية التحتية الجديــــدة التي ينفذها صندوق الاســــتثمارات العامة و برنامج ســــكني الذي يعد من أهم مبادرات وزارة الاســــكان و الــــذي يهدف الــــى زيادة تملــــك المنازل, و أيضــــا المبادرات الحكوميــــة الاخرى التي تهدف الى زيادة المحتوى المحلي.

بالإضافــــة إلــــى ذلــــك، فــــإن زيــــادة الصــــادرات إلــــى البلدان المجاورة، التي تســــتثمر بشكلٍ كبير في البنية التحتية، تمثل فرصــــة واعــــدة للنمــــو، و لتحقيق ذلــــك يجب علــــى أصحاب المصلحة التركيــــز على جودة المنتجــــات، و التغطية الفعالة للأســــواق، و تحقيــــق التكامــــل العكســــي و ضبــــط التكاليف بفعاليــــة مــــع التغلب على هــــذه التحديات و الاســــتفادة من فرص النمو.
